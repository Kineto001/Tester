# netlify.toml
[build]
  # This is the folder Netlify will serve as your static site.
  # We put index.html, css, and js here.
  publish = "static/"
  
[functions]
  # This tells Netlify where to find your serverless functions.
  directory = "netlify/functions/"

# This is the magic part. It rewrites requests from the frontend
# (e.g., /api/generate-test) to execute the correct serverless function.
[[redirects]]
  from = "/api/*"
  to = "/.netlify/functions/:splat"
  status = 200
